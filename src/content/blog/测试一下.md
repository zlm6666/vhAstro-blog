\documentclass[tikz,border=3.14mm]{standalone}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18}
\usepackage{amsmath}
\usepackage{siunitx}

\begin{document}
\begin{tikzpicture}
  \begin{axis}[
      width=12cm,
      height=8cm,
      axis lines = center,
      xlabel = {$x$},
      ylabel = {$y$},
      xmin = -5, xmax = 5,
      ymin = -5, ymax = 5,
      xtick = {-4,-2,...,4},
      ytick = {-4,-2,...,4},
      grid = both,
      grid style = {dashed, gray!30},
      title = {函数图像: \(\sin 20^\circ \cdot x + \cos 20^\circ \cdot y + 1 = 0\)},
      legend pos = north west,
      tick label style = {font=\small},
      label style = {font=\small},
      title style = {font=\small}
    ]

    % 计算三角函数值
    \pgfmathsetmacro{\sinTwenty}{sin(20)}  % 自动转换为弧度
    \pgfmathsetmacro{\cosTwenty}{cos(20)}
    
    % 计算关键点
    \pgfmathsetmacro{\xIntercept}{-1/\sinTwenty}  % y=0时的x值
    \pgfmathsetmacro{\yIntercept}{-1/\cosTwenty}  % x=0时的y值
    
    % 绘制直线
    \addplot [
      domain = -4:4,
      samples = 100,
      color = blue,
      line width = 1.5pt,
      smooth
    ] 
      (x, {(-\sinTwenty*x - 1)/\cosTwenty}) 
      node[pos=0.7, above right, font=\small] {\(y = \frac{-0.3420x - 1}{0.9397}\)};
    
    % 标记关键点
    \addplot [only marks, mark=*, mark size=2pt, color=red] coordinates {
      (0, {\yIntercept})
      ({\xIntercept}, 0)
    };
    
    % 添加关键点标签
    \node[above right, font=\tiny] at (0, {\yIntercept}) 
      {\((0, \num[round-mode=places, round-precision=2]{\yIntercept})\)};
    \node[below left, font=\tiny] at ({\xIntercept}, 0) 
      {\((\num[round-mode=places, round-precision=2]{\xIntercept}, 0)\)};
    
    % 添加图例
    \legend{直线, 关键点}
    
    % 添加方程文本
    \node[align=center, font=\small] at (axis cs:-3,-3) {
      方程: \\
      \(\sin 20^\circ \cdot x + \cos 20^\circ \cdot y + 1 = 0\) \\
      转换后: \\
      \(y = \dfrac{-\sin 20^\circ}{\cos 20^\circ}x - \dfrac{1}{\cos 20^\circ}\) \\
      \(y \approx -0.364x - 1.064\)
    };
    
  \end{axis}
\end{tikzpicture}
\end{document}
